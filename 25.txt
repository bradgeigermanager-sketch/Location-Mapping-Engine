6. Python: core engine function

`python
import math
from typing import Dict, Any, List

def comparepoints10d(pointA: Dict[str, Any],
                       pointB: Dict[str, Any],
                       metric_profile: Dict[str, Any]) -> Dict[str, Any]:
    # 3D
    dx = pointB["coords3d"]["x"] - pointA["coords3d"]["x"]
    dy = pointB["coords3d"]["y"] - pointA["coords3d"]["y"]
    dz = pointB["coords3d"]["z"] - pointA["coords3d"]["z"]

    d3_direction: List[float] = [dx, dy, dz]
    d3_euclidean = math.sqrt(dxdx + dydy + dz*dz)
    d3_manhattan = abs(dx) + abs(dy) + abs(dz)
    d3_chebyshev = max(abs(dx), abs(dy), abs(dz))

    # D4
    d4_relative = abs(pointB["d4"] - pointA["d4"])

    # Brane dims 5–10
    braneA = pointA["brane_dims"]
    braneB = pointB["brane_dims"]

    branestaticdistances: List[float] = []
    branestaticangles: List[float] = []

    for a, b in zip(braneA, braneB):
        dd = abs(b - a)
        da = math.atan(b) - math.atan(a)
        branestaticdistances.append(dd)
        branestaticangles.append(da)

    compositestatic = math.sqrt(sum(dd*dd for dd in branestatic_distances))
    compositeangle = math.sqrt(sum(da*da for da in branestatic_angles))

    w = metric_profile["weights"]
    total_distance = math.sqrt(
        w["gamma3d"]  d3euclidean*2 +
        w["gamma4"]  d4relative*2 +
        w["gammastatic"]  compositestatic*2 +
        w["gammaangle"]  compositeangle*2
    )

    return {
        "pointaid": pointA["point_id"],
        "pointbid": pointB["point_id"],
        "d3euclidean": d3euclidean,
        "d3manhattan": d3manhattan,
        "d3chebyshev": d3chebyshev,
        "d3directionvector": d3_direction,
        "d4relative": d4relative,
        "branestaticdistances": branestaticdistances,
        "branestaticangles": branestaticangles,
        "compositestaticdistance": composite_static,
        "compositeangledistance": composite_angle,
        "totaldistance": totaldistance
    }
